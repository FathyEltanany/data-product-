what's the app about

  1. plotting random points
  2. using shiny ui's gadgets to adjust the appearance of plot

default plot

dataX <- runif(1000)
dataY <- runif(1000)

p<-plot_ly(x = ~dataX, y = ~dataY, type = "scatter")%>%

        layout(
                xaxis = list(title  = "x axis"),
                yaxis = list(title  = "y axis ")
                )
